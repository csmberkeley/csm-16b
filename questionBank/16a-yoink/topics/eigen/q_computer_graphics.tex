    \item Solve for eigenvalues, if any, for the rotation matrix used in part b.

    \ans{
        \begin{center}
            R is the rotation matrix used in part b, \\
            $\begin{bmatrix} 0 & -1 \\ 1 & 0 \end{bmatrix}$ \\
            $R\vec{v}$ = $\lambda\vec{v}$ \\
            det($R - \lambda I$) = 0 \\
            = $R - \lambda I$ = $\begin{bmatrix} 0 - \lambda & -1 \\ 1 & 0 - \lambda \end{bmatrix}$ \\
            = det$\begin{bmatrix} -\lambda & -1 \\ 1 & -\lambda \end{bmatrix}$ = 0 \\
            = $\lambda{^2}$ + 1 = 0 \\ 
            = $\lambda{^2}$ = -1, no real eigenvalues
        \end{center}
    }

    \item The improper rotation matrix $\overline{R}$ is defined by: 
    \begin{center}
        \(\overline{R} = \begin{bmatrix} cos(\theta) & sin(\theta) \\ sin(\theta) & -cos(\theta) \end{bmatrix} \)
    \end{center}
    Solve for eigenvalues of the improper rotation matrix for $\theta$ $\neq$ 0

    \ans{
        \begin{center}
            $\vec{Rv} = \lambda\vec{v}$ \\
            $\det(\vec{R} - \lambda I) = 0$ \\
            = $\vec{R} - \lambda I = \begin{bmatrix} \cos(\theta) - \lambda & \sin(\theta) \\ \sin(\theta) & -\cos(\theta) - \lambda \end{bmatrix}$ \\
            = $\det\begin{bmatrix} \cos(\theta) - \lambda & \sin(\theta) \\ \sin(\theta) & -\cos(\theta) - \lambda \end{bmatrix} = 0$ \\
            = $(\cos \theta - \lambda)(-\cos \theta - \lambda) - \sin^2 \theta = 0$ \\
            = $-\cos^2 \theta - \lambda \cos \theta + \lambda \cos \theta + \lambda^2 - \sin^2 \theta = 0$ \\ 
            $\lambda^2 -1 = 0$ \\
            $\lambda = \pm 1$
        \end{center}
    }
